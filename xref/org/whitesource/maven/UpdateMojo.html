182     @Override
183     public void doExecute() throws MojoExecutionException, MojoFailureException {
184         if (reactorProjects == null) {
185             info("No projects found. Skipping update");
186             return;
187         }
188 
189         // initialize
190         init();
191 
192         // Collect OSS usage information
193         Collection<AgentProjectInfo> projectInfos = extractProjectInfos();
194 
195         // send to white source
196         if (projectInfos == null || projectInfos.isEmpty()) {
197             info("No open source information found.");
198         } else {
199             sendUpdate(projectInfos);
200         }
201     }

203     private void init() {
204         // copy token for modules with special names into moduleTokens.
205         for (Map.Entry<Object, Object> entry : specialModuleTokens.entrySet()) {
206             moduleTokens.put(entry.getKey().toString(), entry.getValue().toString());
207         }
208 
209         // take product name and version from top level project
210         MavenProject topLevelProject = session.getTopLevelProject();
211         if (topLevelProject != null) {
212             if (StringUtils.isBlank(product)) {
213                 product = topLevelProject.getName();
214             }
215             if (StringUtils.isBlank(product)) {
216                 product = topLevelProject.getArtifactId();
217             }
218         }
219     }

221     private Collection<AgentProjectInfo> extractProjectInfos() {
222         Collection<AgentProjectInfo> projectInfos = new ArrayList<AgentProjectInfo>();
223 
224         for (MavenProject project : reactorProjects) {
225             if (shouldProcess(project)) {
226                 projectInfos.add(processProject(project));
227             } else {
228                 info("skipping " + project.getId());
229             }
230         }
231 
232         debugProjectInfos(projectInfos);
233 
234         return projectInfos;
235     }

237     private boolean shouldProcess(MavenProject project) {
238         if (project == null) { return false; }
239 
240         boolean process = true;
241 
242         if (ignorePomModules && "pom".equals(project.getPackaging())) {
243             process = false;
244         } else if (project.equals(mavenProject)) {
245             process = !ignore;
246         } else if (excludes.length > 0 && matchAny(project.getArtifactId(), excludes)) {
247             process = false;
248         } else if (includes.length > 0 && matchAny(project.getArtifactId(), includes)) {
249             process = true;
250         }
251 
252         return process;
253     }
254 
255     private boolean matchAny(String value, String[] patterns) {
256         if (value == null) { return false; }
257 
258         boolean match = false;
259 
260         for (int i=0; i < patterns.length && !match; i++) {
261             String pattern = patterns[i];
262             if (pattern != null)  {
263                 String regex = pattern.replace(".", "\\.").replace("*", ".*");
264                 match = value.matches(regex);
265             }
266         }
267 
268         return match;
269     }

271     private AgentProjectInfo processProject(MavenProject project) {
272         long startTime = System.currentTimeMillis();
273 
274         info("processing " + project.getId());
275 
276         AgentProjectInfo projectInfo = new AgentProjectInfo();
277 
278         // project token
279         if (project.equals(mavenProject)) {
280             projectInfo.setProjectToken(projectToken);
281         } else {
282             projectInfo.setProjectToken(moduleTokens.get(project.getArtifactId()));
283         }
284 
285         // project coordinates
286         projectInfo.setCoordinates(extractCoordinates(project));
287 
288         // parent coordinates
289         if (project.hasParent()) {
290             projectInfo.setParentCoordinates(extractCoordinates(project.getParent()));
291         }
292 
293         // dependencies
294         Map<Dependency, Artifact> lut = createLookupTable(project);
295         for (Dependency dependency : project.getDependencies()) {
296             if (ignoreTestScopeDependencies && Artifact.SCOPE_TEST.equals(dependency.getScope())) {
297                 continue; // exclude test scope dependencies from being sent to the server
298             }
299 
300             DependencyInfo dependencyInfo = getDependencyInfo(dependency);
301 
302             Artifact artifact = lut.get(dependency);
303             if (artifact != null) {
304                 File artifactFile = artifact.getFile();
305                 if (artifactFile != null && artifactFile.exists()) {
306                     try {
307                         dependencyInfo.setSha1(ChecksumUtils.calculateSHA1(artifactFile));
308                     } catch (IOException e) {
309                         debug(Constants.ERROR_SHA1 + " for " + artifact.getId());
310                     }
311                 }
312             }
313 
314             projectInfo.getDependencies().add(dependencyInfo);
315         }
316 
317         info("Total processing time is " + (System.currentTimeMillis() - startTime) + " [msec]");
318 
319         return projectInfo;
320     }

328     private Map<Dependency, Artifact> createLookupTable(MavenProject project) {
329         Map<Dependency, Artifact> lut = new HashMap<Dependency, Artifact>();
330 
331         for (Dependency dependency : project.getDependencies()) {
332             for (Artifact dependencyArtifact : project.getDependencyArtifacts()) {
333                 if (match(dependency, dependencyArtifact)) {
334                     lut.put(dependency, dependencyArtifact);
335                 }
336             }
337         }
338 
339         return lut;
340     }

342     private boolean match(Dependency dependency, Artifact artifact) {
343         boolean match = dependency.getGroupId().equals(artifact.getGroupId()) &&
344                 dependency.getArtifactId().equals(artifact.getArtifactId()) &&
345                 dependency.getVersion().equals(artifact.getVersion());
346 
347         if (match) {
348             if (dependency.getClassifier() == null) {
349                 match = artifact.getClassifier() == null;
350             } else {
351                 match = dependency.getClassifier().equals(artifact.getClassifier());
352             }
353         }
354 
355         if (match) {
356             String type = artifact.getType();
357             if (dependency.getType() == null) {
358                 match = type == null || "jar".equals(type);
359             } else {
360                 match = dependency.getType().equals(type);
361             }
362         }
363 
364         return match;
365     }

367     private DependencyInfo getDependencyInfo(Dependency dependency) {
368         DependencyInfo info = new DependencyInfo();
369 
370         // dependency data
371         info.setGroupId(dependency.getGroupId());
372         info.setArtifactId(dependency.getArtifactId());
373         info.setVersion(dependency.getVersion());
374         info.setScope(dependency.getScope());
375         info.setClassifier(dependency.getClassifier());
376         info.setOptional(dependency.isOptional());
377         info.setType(dependency.getType());
378         info.setSystemPath(dependency.getSystemPath());
379 
380         // exclusions
381         Collection<ExclusionInfo> exclusions = info.getExclusions();
382         for (Exclusion exclusion : dependency.getExclusions()) {
383             exclusions.add(new ExclusionInfo(exclusion.getArtifactId(), exclusion.getGroupId()));
384         }
385 
386         return info;
387     }

389     private void sendUpdate(Collection<AgentProjectInfo> projectInfos) throws MojoFailureException, MojoExecutionException {
390         try {
391             UpdateInventoryResult updateResult;
392             if (checkPolicies) {
393                 info("Checking policies...");
394                 CheckPoliciesResult result = service.checkPolicies(orgToken, product, productVersion, projectInfos);
395 
396                 if (outputDirectory == null ||
397                         (!outputDirectory.exists() && !outputDirectory.mkdirs())) {
398                     warn("Output directory doesn't exist. Skipping policies check report.");
399                 } else {
400                     info("Generating policy check report");
401                     PolicyCheckReport report = new PolicyCheckReport(result);
402                     report.generate(outputDirectory, false);
403                 }
404 
405                 if (result.hasRejections()) {
406                     String msg = "Some dependencies were rejected by the organization's policies.";
407                     error(msg);
408                     throw new MojoExecutionException(msg); // this is handled in base class
409                 } else {
410                     info("All dependencies conform with the organization's policies.");
411                     info("Sending updates to White Source");
412                     updateResult = service.update(orgToken, product, productVersion, projectInfos);
413                 }
414             } else {
415                 info("Sending updates to White Source");
416                 updateResult = service.update(orgToken, product, productVersion, projectInfos);
417             }
418             logResult(updateResult);
419         } catch (WssServiceException e) {
420             throw new MojoExecutionException(Constants.ERROR_SERVICE_CONNECTION + e.getMessage(), e);
421         } catch (IOException e) {
422             throw new MojoExecutionException("Error generating report: " + e.getMessage(), e);
423         }
424     }

426     private void logResult(UpdateInventoryResult result) {
427         info("Inventory update results for " + result.getOrganization());
428 
429         // newly created projects
430         Collection<String> createdProjects = result.getCreatedProjects();
431         if (createdProjects.isEmpty()) {
432             info("No new projects found.");
433         } else {
434             info("Newly created projects:");
435             for (String projectName : createdProjects) {
436                 info(projectName);
437             }
438         }
439 
440         // updated projects
441         Collection<String> updatedProjects = result.getUpdatedProjects();
442         if (updatedProjects.isEmpty()) {
443             info("No projects were updated.");
444         } else {
445             info("Updated projects:");
446             for (String projectName : updatedProjects) {
447                 info(projectName);
448             }
449         }
450     }

452     public void debugProjectInfos(Collection<AgentProjectInfo> projectInfos) {
453         debug("----------------- dumping projectInfos -----------------");
454         debug("Total number of projects : " + projectInfos.size());
455 
456         for (AgentProjectInfo projectInfo : projectInfos) {
457             debug("Project coordiantes: " + projectInfo.getCoordinates().toString());
458             debug("Project parent coordiantes: " + (projectInfo.getParentCoordinates() == null ? "" : projectInfo.getParentCoordinates().toString()));
459             debug("Project token: " + projectInfo.getProjectToken());
460             debug("total # of dependencies: " + projectInfo.getDependencies().size());
461             for (DependencyInfo info : projectInfo.getDependencies()) {
462                 debug(info.toString() + " SHA-1: " + info.getSha1());
463             }
464         }
465 
466         debug("----------------- dump finished -----------------");
467     }
