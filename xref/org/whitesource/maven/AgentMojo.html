202     protected DependencyInfo getDependencyInfo(Dependency dependency) {
203         DependencyInfo info = new DependencyInfo();
204 
205         // dependency data
206         info.setGroupId(dependency.getGroupId());
207         info.setArtifactId(dependency.getArtifactId());
208         info.setVersion(dependency.getVersion());
209         info.setScope(dependency.getScope());
210         info.setClassifier(dependency.getClassifier());
211         info.setOptional(dependency.isOptional());
212         info.setType(dependency.getType());
213         info.setSystemPath(dependency.getSystemPath());
214 
215         // exclusions
216         Collection<ExclusionInfo> exclusions = info.getExclusions();
217         for (Exclusion exclusion : (List<Exclusion>) dependency.getExclusions()) {
218             exclusions.add(new ExclusionInfo(exclusion.getArtifactId(), exclusion.getGroupId()));
219         }
220 
221         return info;

241     protected AgentProjectInfo processProject(MavenProject project) throws MojoExecutionException {
242         long startTime = System.currentTimeMillis();
243 
244         info("processing " + project.getId());
245 
246         AgentProjectInfo projectInfo = new AgentProjectInfo();
247 
248         // project token
249         if (project.equals(mavenProject)) {
250             projectInfo.setProjectToken(projectToken);
251         } else {
252             projectInfo.setProjectToken(moduleTokens.get(project.getArtifactId()));
253         }
254 
255         // project coordinates
256         projectInfo.setCoordinates(extractCoordinates(project));
257 
258         // parent coordinates
259         if (project.hasParent()) {
260             projectInfo.setParentCoordinates(extractCoordinates(project.getParent()));
261         }
262 
263         // dependencies
264         boolean collectDirectDependencies = !resolveInHouseDependencies;
265         if (resolveInHouseDependencies) {
266             try {
267                 info("Getting in-house rules...");
268                 GetInHouseRulesResult inHouseRulesResult = service.getInHouseRules(orgToken);
269                 inHouseRules = inHouseRulesResult.getInHouseRules();
270                 if (inHouseRules.isEmpty()) {
271                     collectDirectDependencies = true;
272                 } else {
273                    projectInfo.getDependencies().addAll(collectDependenciesResolveInHouse(project));
274                 }
275             } catch (WssServiceException e) {
276                 throw new MojoExecutionException(Constants.ERROR_SERVICE_CONNECTION + e.getMessage(), e);
277             } catch (DependencyGraphBuilderException e) {
278                 collectDirectDependencies = true;
279                 projectInfo.getDependencies().clear();
280                 debug("failed to collect dependencies recursively, fallback to direct dependencies only");
281             }
282         }
283 
284         if (collectDirectDependencies) {
285             projectInfo.getDependencies().addAll(collectDependencies(project));
286         }
287 
288         info("Total processing time is " + (System.currentTimeMillis() - startTime) + " [msec]");
289 
290         return projectInfo;
291     }

293     protected Collection<DependencyInfo> collectDependencies(MavenProject project) {
294         Collection<DependencyInfo> dependencyInfos = new ArrayList<DependencyInfo>();
295 
296         Map<Dependency, Artifact> lut = createLookupTable(project);
297         for (Dependency dependency : project.getDependencies()) {
298             if (ignoreTestScopeDependencies && Artifact.SCOPE_TEST.equals(dependency.getScope())) {
299                 continue; // exclude test scope dependencies from being sent to the server
300             }
301 
302             DependencyInfo dependencyInfo = getDependencyInfo(dependency);
303             extractSha1(dependencyInfo, lut.get(dependency));
304             dependencyInfos.add(dependencyInfo);
305         }
306 
307         return dependencyInfos;
308     }

310     protected Collection<DependencyInfo> collectDependenciesResolveInHouse(MavenProject project) throws DependencyGraphBuilderException {
311         // build the dependency graph of the project in order to resolve all transitive dependencies
312         DependencyNode hierarchyRoot = dependencyGraphBuilder.buildDependencyGraph(project, null);
313         Map<Dependency, DependencyNode> dependencyNodeMap = createLookupTable(project, hierarchyRoot);
314         Map<Dependency, Artifact> artifactMap = createLookupTable(project);
315 
316         Collection<DependencyInfo> dependencyInfos = new ArrayList<DependencyInfo>();
317         for (Dependency dependency : project.getDependencies()) {
318             if (ignoreTestScopeDependencies && Artifact.SCOPE_TEST.equals(dependency.getScope())) {
319                 continue; // exclude test scope dependencies from being sent to the server
320             }
321             DependencyInfo dependencyInfo = getDependencyInfo(dependency);
322             dependencyInfos.add(dependencyInfo);
323             extractSha1(dependencyInfo, artifactMap.get(dependency));
324 
325             DependencyNode dependencyNode = dependencyNodeMap.get(dependency);
326             if (dependencyNode != null) {
327                 // resolve in-house dependencies, send them all as flat list (direct dependencies)
328                 if (matchesInHouseRule(dependencyInfo)) {
329                     dependencyInfos.addAll(resolveInHouseDependencies(dependencyNode));
330                 }
331             }
332         }
333 
334         return dependencyInfos;
335     }

337     protected Collection<DependencyInfo> resolveInHouseDependencies(DependencyNode root) {
338         Collection<DependencyInfo> dependencyInfos = new ArrayList<DependencyInfo>();
339 
340         // build maven project from artifact to extract dependencies
341         List<Dependency> dependencies = new ArrayList<Dependency>();
342         Artifact artifact = root.getArtifact();
343         Artifact pomArtifact = artifactFactory.createArtifact(artifact.getGroupId(), artifact.getArtifactId(), artifact.getVersion(), artifact.getScope(), POM);
344         try {
345             MavenProject mavenProject = mavenProjectBuilder.buildFromRepository(pomArtifact, remoteRepos, local);
346             if (mavenProject != null) {
347                 dependencies.addAll(mavenProject.getDependencies());
348             }
349         } catch (ProjectBuildingException e) {
350             debug("error building maven project for " + artifact.getGroupId() + ":" + artifact.getArtifactId() + ":" + artifact.getVersion() + " from repository");
351         }
352 
353         // only collect resolved dependencies (result of the dependencyGraphBuilder)
354         for (DependencyNode childNode : root.getChildren()) {
355             Artifact childArtifact = childNode.getArtifact();
356             for (Dependency dependency : dependencies) {
357                 if (match(dependency, childArtifact)) {
358                     DependencyInfo dependencyInfo = getDependencyInfo(dependency);
359                     dependencyInfos.add(dependencyInfo);
360                     extractSha1(dependencyInfo, childArtifact);
361 
362                     // check in house
363                     if (matchesInHouseRule(dependencyInfo)) {
364                         // recursively collect all dependencies of in-house libraries
365                         dependencyInfos.addAll(resolveInHouseDependencies(childNode));
366                     }
367                     break;
368                 }
369             }
370         }
371 
372         return dependencyInfos;
373     }

375     protected void extractSha1(DependencyInfo dependencyInfo, Artifact artifact) {
376         if (artifact != null) {
377             File artifactFile = artifact.getFile();
378             if (artifactFile != null && artifactFile.exists()) {
379                 try {
380                     dependencyInfo.setSha1(ChecksumUtils.calculateSHA1(artifactFile));
381                 } catch (IOException e) {
382                     debug(Constants.ERROR_SHA1 + " for " + artifact.getId());
383                 }
384             }

388     protected boolean matchesInHouseRule(DependencyInfo dependency) {
389         boolean inHouse = false;
390         if (inHouseRules != null) {
391             for (InHouseRule inHouseRule : inHouseRules) {
392                 String groupId = dependency.getGroupId();
393                 String artifactId = dependency.getArtifactId();
394 
395                 if (StringUtils.isNotBlank(inHouseRule.getNameRegex())) {
396                     if (inHouseRule.match(groupId) || inHouseRule.match(artifactId)) {
397                         debug(dependency.toString() + " matches in-house rule " + inHouseRule.toString());
398                         inHouse = true;
399                         break;
400                     }
401                 } else {
402                     if (inHouseRule.match(groupId, artifactId)) {
403                         debug(dependency.toString() + " matches in-house rule " + inHouseRule.toString());
404                         inHouse = true;
405                         break;
406                     }
407                 }
408             }
409         }
410         return inHouse;
411     }

419     protected Map<Dependency, Artifact> createLookupTable(MavenProject project) {
420         Map<Dependency, Artifact> lut = new HashMap<Dependency, Artifact>();
421 
422         for (Dependency dependency : project.getDependencies()) {
423             for (Artifact dependencyArtifact : project.getDependencyArtifacts()) {
424                 if (match(dependency, dependencyArtifact)) {
425                     lut.put(dependency, dependencyArtifact);
426                 }
427             }
428         }

433     protected Map<Dependency,DependencyNode> createLookupTable(MavenProject project, DependencyNode hierarchyRoot) {
434         Map<Dependency, DependencyNode> lut = new HashMap<Dependency, DependencyNode>();
435 
436         for (Dependency dependency : project.getDependencies()) {
437             for (DependencyNode dependencyNode : hierarchyRoot.getChildren()) {
438                 Artifact dependencyArtifact = dependencyNode.getArtifact();
439                 if (match(dependency, dependencyArtifact)) {
440                     lut.put(dependency, dependencyNode);
441                 }
442             }
443         }

448     protected boolean matchAny(String value, String[] patterns) {
449         if (value == null) { return false; }
450 
451         boolean match = false;
452 
453         for (int i=0; i < patterns.length && !match; i++) {
454             String pattern = patterns[i];
455             if (pattern != null)  {
456                 String regex = pattern.replace(".", "\\.").replace("*", ".*");
457                 match = value.matches(regex);
458             }
459         }
460 
461         return match;
462     }

464     protected boolean match(Dependency dependency, Artifact artifact) {
465         boolean match = dependency.getGroupId().equals(artifact.getGroupId()) &&
466                 dependency.getArtifactId().equals(artifact.getArtifactId()) &&
467                 dependency.getVersion().equals(artifact.getVersion());
468 
469         if (match) {
470             String artifactClassifier = artifact.getClassifier();
471             if (dependency.getClassifier() == null) {
472                 match = artifactClassifier == null || StringUtils.isBlank(artifactClassifier);
473             } else {
474                 match = dependency.getClassifier().equals(artifactClassifier);
475             }
476         }
477 
478         if (match) {
479             String type = artifact.getType();
480             if (dependency.getType() == null) {
481                 match = type == null || "jar".equals(type);
482             } else {
483                 match = dependency.getType().equals(type);
484             }
485         }
486 
487         return match;
488     }

506     protected boolean shouldProcess(MavenProject project) {
507         if (project == null) { return false; }
508 
509         boolean process = true;
510 
511         if (ignorePomModules && POM.equals(project.getPackaging())) {
512             process = false;
513         } else if (project.equals(mavenProject)) {
514             process = !ignore;
515         } else if (excludes.length > 0 && matchAny(project.getArtifactId(), excludes)) {
516             process = false;
517         } else if (includes.length > 0 && matchAny(project.getArtifactId(), includes)) {
518             process = true;
519         }
520 
521         return process;
522     }
